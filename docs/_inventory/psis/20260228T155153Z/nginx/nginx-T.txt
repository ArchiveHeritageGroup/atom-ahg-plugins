# ══════════════════════════════════════════════════════════
# CATCH-ALL: Route ALL remaining requests through Heratio
# ══════════════════════════════════════════════════════════
#
# When .heratio_enabled exists, ALL requests go through
# heratio.php (standalone mode — Symfony not available).
# When absent, $heratio_entry = "index.php" and requests
# go through Symfony as normal.
#
# This MUST be the last location block before the main
# server catch-all, so static files (.css, .js, images)
# are still served directly by nginx.

# Catch-all for /index.php paths (intercepts try_files fallback)
#
# Flow: Request "/admin/secure" → no specific location match →
# main config's `location /` → `try_files $uri /index.php?$args` →
# internal redirect to /index.php → THIS block catches it →
# routes to $heratio_entry (heratio.php when standalone).
#
location ~ ^/index\.php(/|$) {
    include fastcgi_params;
    fastcgi_pass unix:/run/php/php8.3-fpm.sock;
    fastcgi_param SCRIPT_FILENAME $document_root/$heratio_entry;
    fastcgi_param SCRIPT_NAME /$heratio_entry;
    fastcgi_param REQUEST_URI $request_uri;
    fastcgi_read_timeout 300;
}

##
# Registry - registry.theahg.co.za
# Points to AtoM installation, serves ahgRegistryPlugin routes
# Homepage redirects to /registry/
##

server {
    listen 80;
    listen [::]:80;
    server_name registry.theahg.co.za;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name registry.theahg.co.za;

    # Point to the AtoM installation
    root /usr/share/nginx/archive;
    index index.php index.html;

    client_max_body_size 100M;

    ssl_certificate     /etc/letsencrypt/live/theahg.ddns.net-0001/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/theahg.ddns.net-0001/privkey.pem;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;

    access_log /var/log/nginx/registry_access.log;
    error_log  /var/log/nginx/registry_error.log;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header X-Content-Type-Options "nosniff" always;

    # ======================================
    # HOMEPAGE -> redirect to /registry/
    # ======================================
    location = / {
        return 302 /registry/;
    }

    # ======================================
    # REGISTRY PLUGIN STATIC ASSETS
    # ======================================

    # CSS from the plugin
    location ^~ /plugins/ahgRegistryPlugin/css/ {
        alias /usr/share/nginx/archive/plugins/ahgRegistryPlugin/css/;
        expires 7d;
        add_header Cache-Control "public";
    }

    # JS from the plugin
    location ^~ /plugins/ahgRegistryPlugin/js/ {
        alias /usr/share/nginx/archive/plugins/ahgRegistryPlugin/js/;
        expires 7d;
        add_header Cache-Control "public";
    }

    # Registry file uploads (logos, attachments, software packages)
    location ^~ /uploads/registry/ {
        alias /usr/share/nginx/archive/uploads/registry/;
        expires 7d;
        add_header Cache-Control "public";
    }

    # ======================================
    # THEME STATIC ASSETS (needed for layout)
    # ======================================

    # Dominion/AHG theme JS & CSS
    location ^~ /plugins/arDominionB5Plugin/js/dist/ {
        alias /usr/share/nginx/archive/dist/js/;
        try_files $uri =404;
        expires 30d;
    }

    location ^~ /plugins/arDominionB5Plugin/css/dist/ {
        alias /usr/share/nginx/archive/dist/css/;
        try_files $uri =404;
        expires 30d;
    }

    # General static content (css, js, images, fonts)
    location ~* ^/(css|dist|js|images|plugins|vendor|web)/.*\.(css|js|map|png|jpg|jpeg|gif|svg|ico|woff|woff2|ttf|eot|pdf)$ {
        root /usr/share/nginx/archive;
        try_files $uri =404;
        expires 30d;
        add_header Cache-Control "public";
    }

    # Uploads (digital objects, logos)
    location /uploads/ {
        root /usr/share/nginx/archive;
        try_files $uri =404;
        expires 7d;
    }

    # ======================================
    # PHP HANDLING (AtoM Symfony front controller)
    # ======================================
    location ~ ^/(index|qubit_dev)\.php(/|$) {
        include fastcgi_params;
        fastcgi_split_path_info ^(.+?\.php)(/.*)$;
        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
        fastcgi_param PATH_INFO $fastcgi_path_info;
        fastcgi_param PATH_TRANSLATED $document_root$fastcgi_path_info;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_param SCRIPT_NAME $fastcgi_script_name;
        fastcgi_index index.php;
        fastcgi_read_timeout 300;
        fastcgi_buffer_size 128k;
        fastcgi_buffers 4 256k;
        fastcgi_busy_buffers_size 256k;
    }

    # Block direct PHP access (except front controller)
    location ~ \.php$ {
        deny all;
        return 404;
    }

    # Block dotfiles
    location ~ /\.(ht|git|svn) {
        deny all;
    }

    # ======================================
    # MAIN ROUTER (all /registry/* routes go through Symfony)
    # ======================================
    location / {
        try_files $uri /index.php?$args;
    }
}
